void sendActualState(int state) {

  HTTPClient http;

  String url;
  url.reserve(64);  // pre-allocate memory to avoid reallocations
  url = "http://";
  url += server_ip;
  url += ":5000/device/actual_state";

  if (!http.begin(url)) {
    Serial.println("HTTP begin failed");
    return;   // fail fast instead of crashing later
  }

  http.addHeader("Content-Type", "application/json");

  StaticJsonDocument<192> doc;
  doc["actual_state"] = state;

  String body;
  body.reserve(64);
  serializeJson(doc, body);

  int httpCode = http.POST(body);

  if (httpCode > 0) {
    Serial.printf("HTTP POST OK, code: %d\n", httpCode);
  } else {
    Serial.printf("HTTP POST failed, error: %s\n",
                  http.errorToString(httpCode).c_str());
  }

  http.end();
}


void webSocketEvent(WStype_t type, uint8_t* payload, size_t length) {
  if (type == WStype_CONNECTED) {
    StaticJsonDocument<128> doc;
    doc["type"] = "register";
    doc["esp_id"] = esp_id;

    String msg;
    serializeJson(doc, msg);

    webSocket.sendTXT(msg);
  }
  else if (type == WStype_TEXT) {
    StaticJsonDocument<256> doc;
    DeserializationError err = deserializeJson(doc, payload, length);
    if (err) {
      Serial.print("JSON error: ");
      Serial.println(err.c_str());
      return;
    }

    if (doc.containsKey("desired_state")) {
      int desired_state = doc["desired_state"];
      digitalWrite(LED_PIN, desired_state);

      // Optional: defer HTTP POST to loop() for stability
//      sendActualState(desired_state);
    }
  }
}
